## 2. Реализация программы
#include <stdio.h>
#include <locale.h>

#define PI 3.14

int main()
{
	setlocale(LC_ALL, "RUS");
	double D, r, S, V;

	printf("Введите диаметр шара: ");
	scanf("%lf", &D);
	r = D / 2.0;
	S = 4 * PI * r * r;
	V = (4.0 / 3.0) * PI * r * r * r;
	printf("Площадь поверхности шара: %.2f\n", S);
	printf("Объем шара: %.2f\n", V);

	return 0;
}
